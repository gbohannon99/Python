#Here we are defining categories for Master's Degree and Other 
#     From what I researched, the best way to answer this question is 2 group them into categories. Masters and Professional Degrees in one category and Bachelors,Associates, etc in another one
#     Here we are using a dictionary to help define the mapping between the 2 categories.
Categories = {    
                "Masters & Professional Degrees": ["Master’s degree (M.A., M.S., M.Eng., MBA, etc.)", "Professional degree (JD, MD, Ph.D, Ed.D, etc.)"],
                "Undergraduate & Below": ["Bachelor’s degree (B.A., B.S., B.Eng., etc.)", "Primary/elementary school", "Some college/university study without earning a degree",
                                  "Secondary school (e.g. American high school, German Realschule or Gymnasium, etc.)", "Associate degree (A.A., A.S., etc.)", "Something else"]
}

#I know want to make sure that the records im searching for is 'employed full time', so in order to do so I created a list that can be adjusted if you wanted to include those 
#  who are part-time, contractors, etc.
employed_choices = ['Employed, full-time']

#Here is where put all the records that had 'employed, full time' in the employment coloumn into a new dataframe 'employed_df'
employed_df = Education_Level[Education_Level['Employment'].isin(employed_choices)].copy()

#This line of code is needed to categorize the respondents education levels into specific groups as defined in the categories dictionary,
#  which aids in analyzing and visualizing the employment rates based on these 2 education levels 
employed_df['Education Category'] = employed_df['EdLevel'].apply(lambda x: next((key for key, values in Categories.items() if x in values), 'Undergraduate & Below'))

#This line will cound and find the percentage of the 2 education levels
employment_rates = employed_df['Education Category'].value_counts(normalize=True) * 100

#Used to display the percentage
